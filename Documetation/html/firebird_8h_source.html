00284 void motion_set (unsigned char Direction)
00285 {
00286  unsigned char PortARestore = 0;
00287 
00288  Direction &= 0x0F;                     // removing upper nibbel for the protection
00289  PortARestore = PORTA;                  // reading the PORTA original status
00290  PortARestore &= 0xF0;                  // making lower direction nibbel to 0
00291  PortARestore |= Direction;     // adding lower nibbel for forward command and restoring the PORTA status
00292  PORTA = PortARestore;                  // executing the command
00293 }

00349 unsigned char ADC_Conversion(unsigned char Ch)
00350 {
00351         unsigned char a;
00352         if(Ch>7)
00353         {
00354                 ADCSRB = 0x08;
00355         }
00356         Ch = Ch & 0x07;                         
00357         ADMUX= 0x20| Ch;                        
00358         ADCSRA = ADCSRA | 0x40;         //Set start conversion bit
00359         while((ADCSRA&0x10)==0);        //Wait for conversion to complete
00360         a=ADCH;
00361         ADCSRA = ADCSRA|0x10;
00362         ADCSRB = 0x00;
00363         return a;
00364 }
